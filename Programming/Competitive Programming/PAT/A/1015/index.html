    #include "bits/stdc++.h"

using u32 = unsigned;
using i64 = long long;
using u64 = unsigned long long;

constexpr int N = 100000;
std::vector<int> notPrime(N + 1);
std::vector<int> primes;

void euler_sieve() {
    notPrime[0] = 1;
    notPrime[1] = 1;

    for (int i = 2; i <= N; i++) {
        if (!notPrime[i]) {
            primes.push_back(i);
        }

        for (auto p : primes) {
            if (i * p > N) {
                break;
            }

            notPrime[i * p] = 1;

            if (i % p == 0) {
                break;
            }
        }
    }
}

void solve() {
    int n = 0, d = 0;

    while (1) {
        std::cin >> n;

        if (n < 0) {
            return;
        }

        std::cin >> d;

        int nn = n;
        std::vector<int> digits;
        while (nn) {
            digits.push_back(nn % d);
            nn /= d;
        }

        int res = 0;
        for (auto ele : digits) {
            res = res * d + ele;
        }

        std::cout << (!notPrime[n] && !notPrime[res] ? "Yes\n" : "No\n");
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    euler_sieve();
    solve();
    return 0;
}
